## Why this matters

Almost every backend application works with files: reading configuration, parsing uploaded CSVs, writing logs, generating reports, serving static assets. Node's `fs` module is the gateway to all of it. The module exposes three completely different APIs for the same operations — synchronous, callback-based, and promise-based — and knowing which to use and when prevents both blocking your entire server and introducing subtle callback bugs in production.

## Three APIs for the same operations

Node's `fs` module went through two generations of async design, so it now ships three styles side-by-side.

### Synchronous (blocking)

```javascript
const fs = require("fs");

// Blocks the entire Node process until the file is read
const content = fs.readFileSync("./config.json", "utf8");
const config = JSON.parse(content);
console.log(config.port); // 3000
```

`readFileSync` returns the data directly and blocks the event loop until the I/O completes. This is acceptable in two situations: scripts that run once and exit (CLI tools that process files), and application startup code before the server begins accepting requests. Never use synchronous fs calls inside a request handler — they block every other user while one request reads a file.

### Callback-based (legacy async)

```javascript
const fs = require("fs");

fs.readFile("./data.txt", "utf8", (error, content) => {
  if (error) {
    console.error("Could not read file:", error.message);
    return;
  }
  console.log(content);
});

console.log("This runs before the file is read"); // prints first
```

The callback receives `(error, result)` — always check `error` before using `result`. This was the standard pattern from 2009 to roughly 2017. It still works perfectly, but deeply nested callbacks create "callback hell" that is hard to read and harder to debug.

### Promise-based (modern)

```javascript
const fs = require("fs/promises");
// or: const { promises: fs } = require("fs");
// or: import fs from "fs/promises"; // ESM

async function readConfig() {
  try {
    const content = await fs.readFile("./config.json", "utf8");
    return JSON.parse(content);
  } catch (error) {
    console.error("Config read failed:", error.message);
    throw error;
  }
}
```

Use `fs/promises` for all new code. It integrates cleanly with `async/await`, and errors propagate naturally through `try/catch`.

## Common fs operations

```javascript
const fs = require("fs/promises");
const path = require("path");

// Read a file
const content = await fs.readFile("./readme.txt", "utf8");

// Write a file (creates or overwrites)
await fs.writeFile("./output.json", JSON.stringify(data, null, 2), "utf8");

// Append to a file
await fs.appendFile("./app.log", `[${new Date().toISOString()}] Request received\n`);

// List directory contents
const entries = await fs.readdir("./src");
// entries: ['index.js', 'routes', 'utils']

// Get file/directory metadata
const stat = await fs.stat("./package.json");
console.log(stat.isFile()); // true
console.log(stat.isDirectory()); // false
console.log(stat.size); // bytes

// Create a directory (recursive: true creates parent dirs too)
await fs.mkdir("./logs/archive", { recursive: true });

// Delete a file
await fs.unlink("./temp.txt");

// Delete a directory and all contents
await fs.rm("./dist", { recursive: true, force: true });
```

## The path module

File paths work differently on Windows (`\`) and Unix (`/`). The `path` module normalizes this so your code runs on any OS.

```javascript
const path = require("path");

// Join path segments safely (handles OS separators)
const filePath = path.join(__dirname, "data", "users.json");
// /home/user/app/data/users.json (Unix)
// C:\Users\user\app\data\users.json (Windows)

// Resolve to an absolute path
const absolute = path.resolve("./config.json");
// /home/user/app/config.json

// Extract parts of a path
path.dirname("/home/user/app/index.js"); // "/home/user/app"
path.basename("/home/user/app/index.js"); // "index.js"
path.extname("/home/user/app/index.js"); // ".js"

// __dirname is the directory of the current file
// __filename is the full path to the current file
console.log(__dirname); // /home/user/app
```

Always use `path.join` instead of string concatenation for file paths. `/` + path + `/` + file breaks silently on Windows.

## Streams for large files

Reading an entire file into memory with `readFile` works fine for configuration files and small datasets. It breaks down for a 2 GB log file — Node would try to load all 2 GB into RAM simultaneously.

Streams process data in chunks:

```javascript
const fs = require("fs");
const readline = require("readline");

async function processLargeFile(filePath) {
  const fileStream = fs.createReadStream(filePath, { encoding: "utf8" });

  const rl = readline.createInterface({
    input: fileStream,
    crlfDelay: Infinity, // Handle Windows line endings
  });

  let lineCount = 0;

  for await (const line of rl) {
    lineCount++;
    // Process each line without loading the whole file into memory
    if (line.includes("ERROR")) {
      console.error(`Line ${lineCount}: ${line}`);
    }
  }

  console.log(`Processed ${lineCount} lines`);
}
```

`createReadStream` emits data events as chunks arrive from disk. The `readline` interface splits those chunks into lines. This processes a 10 GB file with the same memory footprint as a 10 KB file — the file size no longer determines memory usage.

## Working with JSON files

JSON is the most common file format in Node.js backends:

```javascript
const fs = require("fs/promises");

// Read and parse
async function readJSON(filePath) {
  const content = await fs.readFile(filePath, "utf8");
  return JSON.parse(content);
}

// Write formatted JSON
async function writeJSON(filePath, data) {
  const formatted = JSON.stringify(data, null, 2); // 2-space indent
  await fs.writeFile(filePath, formatted, "utf8");
}

// Usage
const users = await readJSON("./users.json");
users.push({ id: 4, name: "Alice" });
await writeJSON("./users.json", users);
```

For config files that humans edit, pretty-print with `JSON.stringify(data, null, 2)`. For data files only your app reads, skip indentation to save space.

// createFS(files) takes an object mapping paths to content strings.
// Implement: readFile(path), writeFile(path, content),
//            readdir(dir), exists(path), deleteFile(path)
// Throw an error with message 'ENOENT' for missing paths.

function createFS(files) {
// Internal storage — clone so the original object isn't mutated
const store = { ...files };

function readFile(path) {
if (!(path in store)) throw new Error("ENOENT: " + path);
return store[path];
}

function writeFile(path, content) {
store[path] = content;
}

function readdir(dir) {
const prefix = dir.endsWith("/") ? dir : dir + "/";
return Object.keys(store)
.filter((p) => p.startsWith(prefix))
.map((p) => p.slice(prefix.length).split("/")[0])
.filter((name, i, arr) => arr.indexOf(name) === i); // deduplicate
}

function exists(path) {
return path in store;
}

function deleteFile(path) {
if (!(path in store)) throw new Error("ENOENT: " + path);
delete store[path];
}

return { readFile, writeFile, readdir, exists, deleteFile };
}

"/app/utils.js": "module.exports = {}",

## What you learned

- Node's `fs` module has three API styles: synchronous (blocking), callback-based (legacy), and promise-based (`fs/promises`) — use promises for new code
- Common operations: `readFile`, `writeFile`, `appendFile`, `readdir`, `stat`, `mkdir`, `unlink`
- The `path` module normalizes file paths across operating systems — always use `path.join` instead of string concatenation
