## Why this matters

CLI tools are one of Node.js's strongest use cases — they are fast to start, easy to distribute, and integrate naturally with shell scripts and build pipelines. Every developer uses dozens of Node-based CLIs daily: npm, ESLint, TypeScript's `tsc`, Prettier, Vite. Building one from scratch ties together everything you have learned about Node.js: filesystem access, process APIs, error handling, and testable pure functions. The architectural pattern you see here — separate I/O from pure transformation — applies equally to REST APIs and serverless functions.

## Project overview

We are building a CSV processor CLI that:

1. Reads a CSV file specified on the command line
2. Filters rows by a field value
3. Groups remaining rows by a column
4. Aggregates a numeric column per group
5. Writes the result as JSON to an output file

```bash
node process-csv.js --input sales.csv --filter region=west --group product --sum revenue --output report.json
```

## Parsing CLI arguments

Node.js exposes command-line arguments as `process.argv` — an array where index 0 is the `node` executable path, index 1 is the script path, and index 2 onwards are your arguments:

```javascript
// node process-csv.js --input sales.csv --output report.json
console.log(process.argv);
// ['/usr/local/bin/node', '/app/process-csv.js', '--input', 'sales.csv', '--output', 'report.json']
```

Parse them manually:

```javascript
function parseArgs(argv) {
  const args = {};
  const rawArgs = argv.slice(2); // Skip node and script path

  for (let i = 0; i < rawArgs.length; i++) {
    if (rawArgs[i].startsWith("--")) {
      const key = rawArgs[i].slice(2); // Strip leading --
      const value = rawArgs[i + 1]; // Next element is the value
      args[key] = value;
      i++; // Skip the value on the next iteration
    }
  }

  return args;
}
```

For production CLIs, use a library like `commander` or `yargs` — they handle edge cases, generate help text, and validate required arguments. But understanding the raw `process.argv` mechanism first means you are not dependent on the library.

## Reading and writing files

```javascript
const fs = require("fs/promises");
const path = require("path");

async function readInputFile(filePath) {
  const absolutePath = path.resolve(filePath);
  try {
    return await fs.readFile(absolutePath, "utf8");
  } catch (error) {
    if (error.code === "ENOENT") {
      throw new Error(`Input file not found: ${absolutePath}`);
    }
    throw error;
  }
}

async function writeOutputFile(filePath, data) {
  const absolutePath = path.resolve(filePath);
  const directory = path.dirname(absolutePath);
  await fs.mkdir(directory, { recursive: true }); // Create parent dirs if needed
  await fs.writeFile(absolutePath, JSON.stringify(data, null, 2), "utf8");
}
```

## Parsing CSV

CSV (comma-separated values) is simpler than it looks — until you hit quoted fields with commas inside them. For this project we use a basic parser that handles the common case:

```javascript
function parseCSV(text) {
  const lines = text.trim().split("\n");
  if (lines.length === 0) return [];

  const headers = lines[0].split(",").map((h) => h.trim());

  return lines.slice(1).map((line) => {
    const values = line.split(",").map((v) => v.trim());
    const row = {};
    headers.forEach((header, i) => {
      row[header] = values[i] ?? "";
    });
    return row;
  });
}
```

For production CSV with quoted fields, embedded newlines, and different delimiters, use the `csv-parse` library. Never write your own production CSV parser — the edge cases are numerous and tested libraries handle them all.

## The pure transformation function

The most important architectural decision: the core data transformation is a **pure function** with no I/O. It takes data in, returns data out, and has no side effects. This makes it trivially testable.

```javascript
function processRecords(records, { filterField, filterValue, groupBy, sumField }) {
  // Step 1: Filter
  const filtered = filterField
    ? records.filter((row) => row[filterField] === filterValue)
    : records;

  // Step 2: Group and aggregate
  const groups = {};
  for (const row of filtered) {
    const key = row[groupBy] ?? "unknown";
    const value = parseFloat(row[sumField]) || 0;
    groups[key] = (groups[key] || 0) + value;
  }

  return groups;
}
```

This function is deterministic, has no I/O, and can be unit tested without touching the filesystem:

```javascript
const records = [
  { region: "west", product: "Widget", revenue: "100" },
  { region: "west", product: "Gadget", revenue: "200" },
  { region: "east", product: "Widget", revenue: "150" },
];

const result = processRecords(records, {
  filterField: "region",
  filterValue: "west",
  groupBy: "product",
  sumField: "revenue",
});
// { Widget: 100, Gadget: 200 }
```

## Wiring it together and handling errors

```javascript
async function main() {
  const args = parseArgs(process.argv);

  if (!args.input || !args.output) {
    console.error("Usage: node process-csv.js --input <file> --output <file>");
    console.error("       --filter <field>=<value> --group <field> --sum <field>");
    process.exit(1); // Exit code 1 signals failure
  }

  let filterField, filterValue;
  if (args.filter) {
    [filterField, filterValue] = args.filter.split("=");
  }

  try {
    const csvText = await readInputFile(args.input);
    const records = parseCSV(csvText);

    // Progress logging to stderr (not stdout — keeps stdout clean for piping)
    console.error(`Loaded ${records.length} records`);

    const result = processRecords(records, {
      filterField,
      filterValue,
      groupBy: args.group,
      sumField: args.sum,
    });

    await writeOutputFile(args.output, result);
    console.error(`Written to ${args.output}`);
    process.exit(0); // Exit code 0 signals success
  } catch (error) {
    console.error("Error:", error.message);
    process.exit(1);
  }
}

main();
```

## Making it executable

To run as `./process-csv.js` instead of `node process-csv.js`, add a shebang line:

```javascript
#!/usr/bin/env node
// ^ This tells the OS to use node from PATH to execute this file

const { parseArgs, parseCSV, processRecords } = require("./lib");
// ... rest of the script
```

Then make it executable:

```bash
chmod +x process-csv.js

# Now you can run it directly
./process-csv.js --input data.csv --output result.json
```

The shebang `#!/usr/bin/env node` is the standard approach — it finds `node` on the user's PATH rather than hardcoding the path (`/usr/local/bin/node` would break on systems with nvm or different installations).

## Exit codes

Exit codes communicate success or failure to the shell and CI systems:

| Code  | Meaning                             |
| ----- | ----------------------------------- |
| `0`   | Success                             |
| `1`   | General error                       |
| `2`   | Misuse of command (wrong arguments) |
| `126` | Permission denied                   |
| `127` | Command not found                   |

Shell scripts and CI pipelines check exit codes. `process.exit(1)` causes `npm run` to fail, which stops a CI pipeline. This is the correct behavior — a failed tool should fail loudly.

## What you learned

- `process.argv` exposes CLI arguments as an array; `process.argv.slice(2)` gives your actual flags and values
- Exit code `0` signals success, `1` signals failure — CI systems and shell scripts depend on these codes
- Separating pure data transformation functions from I/O makes code testable without touching the filesystem
