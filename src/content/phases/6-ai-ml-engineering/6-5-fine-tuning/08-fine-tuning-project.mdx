## Why this matters

Individual concepts — data preparation, LoRA configuration, training loops, evaluation, deployment — are useful in isolation. But fine-tuning is an end-to-end pipeline where each stage feeds the next. A mistake in data preparation surfaces as poor evaluation metrics. A wrong LoRA rank manifests as overfitting during training. This lesson connects every concept from this module into a single working project you can adapt to your own tasks.

## The project: a code review model

We are building a fine-tuned model that reviews JavaScript code and returns structured JSON output with identified issues, severity levels, and actionable suggestions. This is a realistic task where fine-tuning makes sense — the base model can already review code, but we need it to consistently output a specific JSON format with our severity taxonomy.

## Stage 1: Data preparation

Every fine-tuning project starts with data. We need system/user/assistant message triples in JSONL format.

```javascript
// Stage 1: Build and validate the training dataset
const SYSTEM_PROMPT =
  "You are a JavaScript code reviewer. For each code snippet, respond with a JSON object containing: issues (array of strings), severity (low|medium|high|critical), and suggestion (string with actionable fix).";

function createTrainingExample(code, issues, severity, suggestion) {
  return {
    messages: [
      { role: "system", content: SYSTEM_PROMPT },
      { role: "user", content: `Review this code:\n\`\`\`javascript\n${code}\n\`\`\`` },
      {
        role: "assistant",
        content: JSON.stringify({ issues, severity, suggestion }),
      },
    ],
  };
}

// Build the dataset — in production this comes from human-reviewed logs
const dataset = [
  createTrainingExample(
    "let x = null;\nx.toString();",
    ["null dereference on line 2"],
    "high",
    "Add a null check before calling methods: if (x !== null) { x.toString(); }"
  ),
  createTrainingExample(
    "const password = 'admin123';",
    ["hardcoded credential"],
    "critical",
    "Move secrets to environment variables: const password = process.env.DB_PASSWORD;"
  ),
  createTrainingExample(
    "function add(a, b) {\n  return a + b;\n}",
    ["no parameter type validation", "no JSDoc"],
    "low",
    "Add type checks and documentation for maintainability."
  ),
  createTrainingExample(
    "eval(userInput);",
    ["eval with untrusted input", "code injection vulnerability"],
    "critical",
    "Never use eval with user input. Use JSON.parse() for data or a sandboxed interpreter."
  ),
  createTrainingExample(
    "for (let i = 0; i <= arr.length; i++) {\n  console.log(arr[i]);\n}",
    ["off-by-one error: <= should be <"],
    "high",
    "Change i <= arr.length to i < arr.length to avoid accessing undefined."
  ),
];

// Validate every example
function validateExample(example) {
  const errors = [];
  if (example.messages.length !== 3) errors.push("Must have exactly 3 messages");
  if (example.messages[0].role !== "system") errors.push("First message must be system");
  if (example.messages[1].role !== "user") errors.push("Second message must be user");
  if (example.messages[2].role !== "assistant") errors.push("Third message must be assistant");

  try {
    const parsed = JSON.parse(example.messages[2].content);
    if (!Array.isArray(parsed.issues)) errors.push("Assistant response missing issues array");
    if (!["low", "medium", "high", "critical"].includes(parsed.severity)) {
      errors.push("Invalid severity: " + parsed.severity);
    }
    if (typeof parsed.suggestion !== "string") errors.push("Missing suggestion string");
  } catch {
    errors.push("Assistant response is not valid JSON");
  }

  return errors;
}

let validCount = 0;
dataset.forEach((example, i) => {
  const errors = validateExample(example);
  if (errors.length === 0) {
    validCount++;
  } else {
    console.log(`Example ${i}: ${errors.join(", ")}`);
  }
});
console.log(`\nValidation: ${validCount}/${dataset.length} examples pass`);

// Output JSONL
const jsonl = dataset.map((ex) => JSON.stringify(ex)).join("\n");
console.log(`\nJSONL output (${jsonl.split("\n").length} lines, ${jsonl.length} bytes)`);
```

## Stage 2: LoRA configuration

With validated data, we configure the training parameters. The choices here depend on dataset size, task complexity, and available hardware.

```javascript
// Stage 2: Configure LoRA and training parameters
function buildTrainingConfig(datasetSize, gpuMemoryGB) {
  // Adapt configuration to dataset size and hardware
  const isSmallDataset = datasetSize < 200;
  const isLimitedGPU = gpuMemoryGB <= 24;

  const config = {
    // Model selection
    baseModel: isLimitedGPU ? "mistralai/Mistral-7B-v0.3" : "meta-llama/Llama-3.1-8B",
    quantization: isLimitedGPU ? "4bit" : "none",

    // LoRA parameters — conservative for small datasets
    lora: {
      rank: isSmallDataset ? 8 : 16,
      alpha: isSmallDataset ? 16 : 32,
      targetModules: ["q_proj", "k_proj", "v_proj", "o_proj"],
      dropout: isSmallDataset ? 0.1 : 0.05, // More regularization for small data
    },

    // Training parameters
    training: {
      epochs: isSmallDataset ? 2 : 3,
      batchSize: isLimitedGPU ? 2 : 4,
      gradientAccumulation: isLimitedGPU ? 8 : 4,
      learningRate: 2e-4,
      warmupRatio: 0.03,
      scheduler: "cosine",
      maxSeqLength: 1024,
    },

    // Evaluation
    evaluation: {
      strategy: "steps",
      evalSteps: 25,
      saveSteps: 25,
      loadBestAtEnd: true,
      metric: "eval_loss",
    },
  };

  const effectiveBatch = config.training.batchSize * config.training.gradientAccumulation;
  const stepsPerEpoch = Math.ceil((datasetSize * 0.9) / effectiveBatch); // 90% for training
  const totalSteps = stepsPerEpoch * config.training.epochs;

  console.log("Training Configuration:");
  console.log(`  Base model: ${config.baseModel}`);
  console.log(`  Quantization: ${config.quantization}`);
  console.log(`  LoRA rank: ${config.lora.rank}, alpha: ${config.lora.alpha}`);
  console.log(`  Effective batch: ${effectiveBatch}`);
  console.log(`  Total steps: ${totalSteps}`);
  console.log(`  Eval every: ${config.evaluation.evalSteps} steps`);

  return config;
}

const config = buildTrainingConfig(500, 24);
```

## Stage 3: Training simulation

In production, training runs on a GPU. Here we simulate the training loop to understand what happens at each step and what to watch for.

```javascript
// Stage 3: Simulate the training process
function simulateTraining(datasetSize, config) {
  const trainSize = Math.floor(datasetSize * 0.9);
  const effectiveBatch = config.training.batchSize * config.training.gradientAccumulation;
  const stepsPerEpoch = Math.ceil(trainSize / effectiveBatch);

  let trainLoss = 3.2;
  let bestValLoss = Infinity;
  let bestStep = 0;
  let patience = 0;
  const maxPatience = 3;
  const history = [];

  console.log("Starting training...\n");

  for (let epoch = 1; epoch <= config.training.epochs; epoch++) {
    for (let step = 1; step <= stepsPerEpoch; step++) {
      const globalStep = (epoch - 1) * stepsPerEpoch + step;

      // Simulate loss decrease with noise
      const progress = globalStep / (stepsPerEpoch * config.training.epochs);
      trainLoss = 3.2 * Math.exp(-3 * progress) + 0.3 + (Math.random() - 0.5) * 0.1;

      // Evaluate at intervals
      if (globalStep % config.evaluation.evalSteps === 0) {
        // Validation loss — starts rising after ~60% progress (overfitting)
        const valLoss = trainLoss + 0.1 + (progress > 0.6 ? (progress - 0.6) * 2 : 0);

        history.push({ step: globalStep, trainLoss, valLoss, epoch });

        const improved = valLoss < bestValLoss;
        if (improved) {
          bestValLoss = valLoss;
          bestStep = globalStep;
          patience = 0;
        } else {
          patience++;
        }

        console.log(
          `Step ${globalStep} (epoch ${epoch}) | Train: ${trainLoss.toFixed(3)} | Val: ${valLoss.toFixed(3)} | ${improved ? "IMPROVED" : "no improvement (" + patience + "/" + maxPatience + ")"}`
        );

        if (patience >= maxPatience) {
          console.log(`\nEarly stopping at step ${globalStep}`);
          console.log(`Best checkpoint: step ${bestStep} (val loss: ${bestValLoss.toFixed(3)})`);
          return { bestStep, bestValLoss, history };
        }
      }
    }
  }

  console.log(`\nTraining complete. Best: step ${bestStep} (val loss: ${bestValLoss.toFixed(3)})`);
  return { bestStep, bestValLoss, history };
}

const trainingResult = simulateTraining(500, config);
```

## Stage 4: Evaluation

Load the best checkpoint and run it against the evaluation dataset. We check both format compliance and task accuracy.

```javascript
// Stage 4: Evaluate the best checkpoint
function evaluateCheckpoint(evalExamples) {
  const metrics = {
    formatCompliance: 0,
    severityCorrect: 0,
    issuesFound: 0,
    totalIssues: 0,
    total: evalExamples.length,
  };

  for (const example of evalExamples) {
    // Simulate model output (in practice, run inference on eval set)
    const modelOutput = example.simulatedOutput;

    // Check format
    try {
      const parsed = JSON.parse(modelOutput);
      if (parsed.issues && parsed.severity && parsed.suggestion) {
        metrics.formatCompliance++;
      }

      // Check severity
      if (parsed.severity === example.expectedSeverity) {
        metrics.severityCorrect++;
      }

      // Check issue detection
      const foundCount = parsed.issues.filter((issue) =>
        example.expectedIssues.some((expected) =>
          issue.toLowerCase().includes(expected.toLowerCase())
        )
      ).length;
      metrics.issuesFound += foundCount;
      metrics.totalIssues += example.expectedIssues.length;
    } catch {
      // Format failure — counts against format compliance
    }
  }

  return {
    formatCompliance: ((metrics.formatCompliance / metrics.total) * 100).toFixed(1),
    severityAccuracy: ((metrics.severityCorrect / metrics.total) * 100).toFixed(1),
    issueRecall: ((metrics.issuesFound / metrics.totalIssues) * 100).toFixed(1),
  };
}

const evalExamples = [
  {
    simulatedOutput:
      '{"issues":["null dereference"],"severity":"high","suggestion":"Add null check."}',
    expectedSeverity: "high",
    expectedIssues: ["null"],
  },
  {
    simulatedOutput:
      '{"issues":["hardcoded password"],"severity":"critical","suggestion":"Use env vars."}',
    expectedSeverity: "critical",
    expectedIssues: ["hardcoded"],
  },
  {
    simulatedOutput:
      '{"issues":["no validation"],"severity":"medium","suggestion":"Add input checks."}',
    expectedSeverity: "medium",
    expectedIssues: ["validation", "type check"],
  },
  {
    simulatedOutput:
      '{"issues":["eval injection"],"severity":"critical","suggestion":"Remove eval."}',
    expectedSeverity: "critical",
    expectedIssues: ["eval"],
  },
];

const evalResults = evaluateCheckpoint(evalExamples);
console.log("\nEvaluation Results:");
console.log(`  Format compliance: ${evalResults.formatCompliance}%`);
console.log(`  Severity accuracy: ${evalResults.severityAccuracy}%`);
console.log(`  Issue recall: ${evalResults.issueRecall}%`);

const isReadyForDeploy =
  parseFloat(evalResults.formatCompliance) >= 95 && parseFloat(evalResults.severityAccuracy) >= 80;
console.log(`\nDeploy decision: ${isReadyForDeploy ? "READY" : "NEEDS MORE WORK"}`);
```

## Stage 5: Deployment

With a validated model, we configure serving and build the API integration.

```javascript
// Stage 5: Deployment configuration
function generateDeploymentPlan(evalResults, modelConfig) {
  const plan = {
    steps: [
      {
        name: "Merge LoRA adapter",
        command:
          "python merge_adapter.py --base-model " +
          modelConfig.baseModel +
          " --adapter ./output/best-checkpoint --output ./merged-model",
        purpose: "Combine adapter weights into base model for zero-overhead inference",
      },
      {
        name: "Quantize for serving",
        command:
          "python quantize.py --model ./merged-model --format awq --bits 4 --output ./quantized-model",
        purpose: "Reduce model size by ~4x with minimal quality loss",
      },
      {
        name: "Start vLLM server",
        command:
          "vllm serve ./quantized-model --port 8000 --max-model-len 1024 --tensor-parallel-size 1",
        purpose: "Launch high-throughput serving with PagedAttention",
      },
      {
        name: "Health check",
        command: "curl http://localhost:8000/health",
        purpose: "Verify the server is ready to accept requests",
      },
    ],

    apiEndpoint: {
      url: "http://localhost:8000/v1/chat/completions",
      method: "POST",
      headers: { "Content-Type": "application/json" },
      exampleBody: {
        model: "code-reviewer",
        messages: [
          { role: "system", content: "You are a JavaScript code reviewer. Respond in JSON." },
          { role: "user", content: "Review: eval(input)" },
        ],
        max_tokens: 300,
        temperature: 0.1,
      },
    },
  };

  console.log("Deployment Plan:");
  plan.steps.forEach((step, i) => {
    console.log(`\n  ${i + 1}. ${step.name}`);
    console.log(`     ${step.purpose}`);
    console.log(`     $ ${step.command}`);
  });

  console.log("\n\nAPI Usage:");
  console.log(`  POST ${plan.apiEndpoint.url}`);
  console.log(`  Body: ${JSON.stringify(plan.apiEndpoint.exampleBody, null, 2)}`);

  return plan;
}

generateDeploymentPlan(evalResults, config);
```

## What you learned

- A fine-tuning project follows five stages: **data preparation, configuration, training, evaluation, and deployment** — each feeding into the next.
- **Data and evaluation** consume the majority of effort (~60%) — the training run itself is the automated part.
- **Deploy readiness** is determined by evaluation metrics on held-out data, not training loss — merge the adapter, quantize for efficiency, and serve behind an OpenAI-compatible API.
